### Область видимости переменных

- Глобальный контекст
- Локальный контекст

#### global
```python
# Глобальная переменная, определена вне любой из функций и доступна любой функции 
global_name1 = "Tom"
global_name2 = "Tom"

def say_hi():
    print("Hello", global_name1)

def say_bye():
    print("Good bye", global_name1)
# ------------------------------
def say_hi():
    local_name = "Sam"  # лакальная переменная, определяется внутри функции и доступна только в этой функции
    
    # скрытие глобальной переменной (объявлена локальная с таким же именем, и далее глобальная не видна)
    global_name1 = "Bag" 
    print(local_name, global_name1)
    
    # мы хотим изменить в локальной функции глобальную переменную, а не крыть её
    global global_name2
    global_name2 = "OK" # будет изменена глобальная переменная
    print( local_name, global_name2)
```
#### nonlocal

```python
# Переменная n определена во внешней функции outer().
# локальная функция inner() выводит на консоль значение переменной n, 
# Затем в функции outer() вызывается внутренняя функция inner().

def outer1():  # внешняя функция
    n = 5
 
    def inner():    # вложенная функция
        print(n)
 
    inner()     # 5
    print(n)
 
 
outer1()     # 5

# Теперь возьмем другую ситуацию, когда вложенная функция присваивает значение переменной.
# При присвоении значения будет создана новая переменная n, которая скроет переменную n из окружающей внешней функции outer. 
def outer2():  # внешняя функция
    n = 5
 
    def inner():    # вложенная функция
        n = 25
        print(n)
 
    inner()     # 25
    print(n)
 
 
outer2()     # 5 


# Чтобы во вложенной функции указать, 
# что идентификатор во вложенной функции будет представлять 
# переменную из окружающей функции, 
# применяется выражение nonlocal:

def outer3():  # внешняя функция
    n = 5
 
    def inner():    # вложенная функция
        nonlocal n  # указываем, что n - это переменная из окружающей функции
        n = 25
        print(n)
 
    inner()     # 25
    print(n)
 
outer3()          # 25
```
